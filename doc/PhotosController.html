def update
  @photo =  Photo.find(params[:id])
  respond_to do |format|
    if @photo.update_attributes(params[:photo])
      @photo.original_picture.reprocess!
      format.html {redirect_to photo_path(@photo) }
    else
      format.html { render action: "edit" }
    end
  end
end
